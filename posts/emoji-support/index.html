\section{Fisher线性判别}

\subsection{符号说明}

{\centering$X=(x_1\ x_2\ \dots\ x_N)^T=\left(\begin{array}{c} x_1^T \ x_2^T \ \vdots \ x_N^T \end{array}\right)_{N \times p}$

$Y=(y_1\ y_2\ \dots\ y_N)^T=\left(\begin{array}{c} y_1 \ y_2 \ \vdots \ y_N \end{array}\right)_{N \times 1}$

$x_i \in \mathbb{R}^p$, $y_i \in {-1,1}$

$x_{c_1}={x_i|y_i=1}$, $x_{c_2}={x_i|y_i=-1}$

$|x_{c_1}|=N_1$, $|x_{c_2}|=N_2$, $N_1+N_2=N$

}

\subsection{基本思想}

寻找一个最合适的投影轴，目标是使得：

\textbf{两类样本在该轴上投影之间的距离尽可能远；每一类样本的投影尽可能紧凑。}

\subsection{定义目标函数}

我们假设投影轴的方向向量为$w$，则投影后的样本点为

\begin{equation} z_i=w^Tx_i \end{equation}

投影后的样本均值为

\begin{equation} \bar{z}=\frac{1}{N}\sum\limits_{i=1}^{N}z_i=\frac{1}{N}\sum\limits_{i=1}^{N}w^Tx_i \end{equation}

投影后的样本类内离散度矩阵为

\begin{equation} S_z=\frac{1}{N}\sum\limits_{i=1}^N(z_i-\bar{z})(z_i-\bar{z})^T=\frac{1}{N}\sum\limits_{i=1}^N(w^Tx_i-\bar{z})(w^Tx_i-\bar{z})^T \end{equation}

具体来说：

\begin{equation} c_1: \bar{z_1}=\frac{1}{N_1}\sum\limits_{i=1}^{N_1}w^Tx_i \end{equation}

\begin{equation} S_{1}=\frac{1}{N_1}\sum\limits_{i=1}^{N_1}(w^Tx_i-\bar{z_1})(w^Tx_i-\bar{z_1})^T % \end{aligned} \end{equation}

\begin{equation} c_2: \bar{z_2}=\frac{1}{N_2}\sum\limits_{i=1}^{N_2}w^Tx_i \end{equation}

\begin{equation} S_{2}=\frac{1}{N_2}\sum\limits_{i=1}^{N_2}(w^Tx_i-\bar{z_2})(w^Tx_i-\bar{z_2})^T \end{equation}

类间：$(\bar{z_1}-\bar{z_2})^2$

类内：$S_1+S_2$

要让类间距离最大，类内距离最小，可以定义目标函数为：

\begin{equation} J(w)=(\bar{z_1}-\bar{z_2})^2/(S_1+S_2) \end{equation}

\begin{equation} \hat{w}=arg\max\limits_{w}J(w) \end{equation}

对于$J(w)$：

\begin{equation} \begin{aligned} \mbox{分子}&=(\bar{z_1}-\bar{z_2})^2\ &=(\frac{1}{N_1}\sum\limits_{i=1}^{N_1}w^Tx_i-\frac{1}{N_2}\sum\limits_{i=1}^{N_2}w^Tx_i)^2\ &=[w^T(\frac{1}{N_1}\sum\limits_{i=1}^{N_1}x_i-\frac{1}{N_2}\sum\limits_{i=1}^{N_2}x_i)]^2\ &=[w^T(\bar{x_{c_1}}-\bar{x_{c_2}})]^2\ &=w^T(\bar{x_{c_1}}-\bar{x_{c_2}})(\bar{x_{c_1}}-\bar{x_{c_2}})^Tw \end{aligned} \end{equation}

\begin{equation} \begin{aligned} S_1&=\frac{1}{N_1}\sum\limits_{i=1}^{N_1}(w^Tx_i-\bar{z_1})(w^Tx_i-\bar{z_1})^T\ &=\frac{1}{N_1}\sum\limits_{i=1}^{N_1}(w^Tx_i-\frac{1}{N}\sum\limits_{j=1}^{N}w^Tx_j)(w^Tx_i-\frac{1}{N}\sum\limits_{j=1}^{N}w^Tx_j)^T\ &=\frac{1}{N_1}\sum\limits_{i=1}^{N_1}(w^Tx_i-w^T\bar{x_{c_1}})(w^Tx_i-w^T\bar{x_{c_1}})^T\ &=\frac{1}{N_1}\sum\limits_{i=1}^{N_1}w^T(x_i-\bar{x_{c_1}})(x_i-\bar{x_{c_1}})^Tw\ &=w^T\underbrace{\frac{1}{N_1}\sum\limits_{i=1}^{N_1}(x_i-\bar{x_{c_1}})(x_i-\bar{x_{c_1}})^T}{\mbox{定义为}S{c_1}}w\ &=w^TS_{c_1}w \end{aligned} \end{equation}

\begin{equation} \begin{aligned} \mbox{分母}&=(S_1+S_2)\ &=w^TS_{c_1}w+w^TS_{c_2}w\ &=w^T(S_{c_1}+S_{c_2})w \end{aligned} \end{equation}

因此，$J(w)$可以写成：

\begin{equation} J(w)=\frac{w^T(\bar{x_{c_1}}-\bar{x_{c_2}})(\bar{x_{c_1}}-\bar{x_{c_2}})^Tw}{w^T(S_{c_1}+S_{c_2})w} \end{equation}

\subsection{闭式解}

Fisher判别准则为：

\begin{equation} \hat{w}=arg\max\limits_{w}J(w)=arg\max\limits_{w}\frac{w^T(\bar{x_{c_1}}-\bar{x_{c_2}})(\bar{x_{c_1}}-\bar{x_{c_2}})^Tw}{w^T(S_{c_1}+S_{c_2})w} \end{equation}

采用拉格朗日乘子法：

\begin{equation} L(w,\lambda)=w^T(\bar{x_{c_1}}-\bar{x_{c_2}})(\bar{x_{c_1}}-\bar{x_{c_2}})^Tw-\lambda(w^T(S_{c_1}+S_{c_2})w-c) \end{equation}

对$w$求导：

\begin{equation} \frac{\partial L(w,\lambda)}{\partial w}=(\bar{x_{c_1}}-\bar{x_{c_2}})(\bar{x_{c_1}}-\bar{x_{c_2}})^Tw-\lambda(S_{c_1}+S_{c_2})w\overset{\mbox{令}}{=}0 \end{equation}

若$S_{c_1}+S_{c_2}$可逆，则有：

\begin{equation} \lambda w^*=(S_{c_1}+S_{c_2})^{-1}(\bar{x_{c_1}}-\bar{x_{c_2}})(\bar{x_{c_1}}-\bar{x_{c_2}})^Tw \end{equation}

又$(\bar{x_{c_1}}-\bar{x_{c_2}})^Tw$为标量，因此有：

\begin{equation} w^*=(S_{c_1}+S_{c_2})^{-1}(\bar{x_{c_1}}-\bar{x_{c_2}}) \end{equation}

\subsection{决策面}

在正态分布及等协方差矩阵的假设下, 以$\bar{x_{c_i}}$估计均值，$S_{c_i}$估计协方差矩阵，则有： \begin{equation} \begin{aligned} & w^*=(S_{c_1}+S_{c_2})^{-1}(\bar{x_{c_1}}-\bar{x_{c_2}})\ & w_0=\frac{1}{2}(\bar{x_{c_1}}+\bar{x_{c_2}})^T(S_{c_1}+S_{c_2})^{-1}(\bar{x_{c_1}}-\bar{x_{c_2}})+\ln\frac{P(c_2)}{P(c_1)} \end{aligned} \end{equation}

% 当$d$ 和 $N$ 很大时, $y$ 近似正态分布, 可在 $\mathbf{Y}$ 空间 内用 Bayes 分类器;

利用先验知识确定一维分类阈值 $w_0$ % \begin{equation} % y_0=\frac{\tilde{m}_1+\tilde{m}_2}{2} ; \quad y_0=\frac{N_1 \tilde{m}_1+N_2 \tilde{m}_2}{N_1+N_2}=\tilde{m} ; \quad y_0=\frac{\tilde{m}_1+\tilde{m}_2}{2}+\frac{\ln \left(P\left(\omega_1 / \omega_2\right)\right)}{N_1+N_2-2} ; % \end{equation}

$$ \begin{aligned} & w_0=\frac{w^{*T}(\bar{x_{c_1}}+\bar{x_{c_2}})}{2}\ & w_0=w^{*T}\bar{x}\ \end{aligned} $$

决策面为：

$$ w^{*T}x=w_0 $$
